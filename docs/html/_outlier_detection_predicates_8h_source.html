139  Norm norm;
140  auto point1 = std::get<ProbeTraits::ProbeColumns::PROJECTED_COORDS>(p1);
141  auto point2 = std::get<ProbeTraits::ProbeColumns::PROJECTED_COORDS>(p2);
142  Vector v = point1 - point2;
143  auto t1 = std::get<ProbeTraits::ProbeColumns::SAMPLE_DATE>(p1);
144  auto t2 = std::get<ProbeTraits::ProbeColumns::SAMPLE_DATE>(p2);
145  NT tdiff = abs(static_cast<NT>(t1) - static_cast<NT>(t2));
146  assert(tdiff > 0);
147  NT squared_distance = norm(v);
148  NT squared_speed = squared_distance / (tdiff * tdiff);
149  return squared_speed <= squared_threshold;
