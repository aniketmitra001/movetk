154  template<class RowIterator, class ColumnIterator, class OutputIterator,
155  typename = movetk_core::requires_random_access_iterator<RowIterator>,

161  void operator()(RowIterator rfirst, RowIterator rbeyond,
162  ColumnIterator cfirst, ColumnIterator cbeyond, OutputIterator result) {
163  LogLikelihood ll;
164 
165  std::size_t num_rows = std::distance(rfirst, rbeyond);
166  std::size_t num_columns = std::distance(cfirst, cbeyond);
167 
168  Row dp_row;
169  Table dp_table (num_rows);
170 
171  // Fill up Dynamic Programming Table
172  ll(*rfirst, cfirst, cbeyond, movetk_core::movetk_back_insert_iterator(dp_row));
173 
174  std::transform(std::begin(dp_row), std::end(dp_row), std::begin(dp_row),ic);
175  auto min_ic = std::min_element(std::cbegin(dp_row), std::cend(dp_row));
176  std::size_t i = 0;
177  dp_table[i] = std::move(dp_row);
178  i++;
179  RowIterator rit = rfirst + 1;
180  while(rit != rbeyond){
181  ColumnIterator cit = cfirst;
182  std::size_t j = 0;
183  Row dp_row (num_columns);
184  while(cit != cbeyond){
185  NT log_likelihood = ll(*rit, *cit);
186  NT ic_extend = dp_table[i - 1][j];
187  NT ic_append = *min_ic + ic.penalty_factor;
188  dp_row[j] = min(ic_extend, ic_append) - 2 * log_likelihood;
189  cit++;
190  j++;
191  }
192  min_ic = std::min_element(std::cbegin(dp_row), std::cend(dp_row));
193  dp_table[i] = std::move(dp_row);
194  rit++;
195  i++;
196  }
197 
198  // Backtrack through the Dynamic Programming Table
199  TableIterator table_iterator = std::cend(dp_table) - 1;
200  RowIterator parameter_iterator = rbeyond - 1;
201  auto last_min_ic = std::min_element(std::cbegin(*table_iterator), std::cend(*table_iterator));
202  std::size_t last_position = std::distance(std::cbegin(*table_iterator), last_min_ic);
203  *result = parameter_iterator;
204 
205  if (num_rows == 1)
206  return;
207 
208  table_iterator--;
209  parameter_iterator--;
210 
211  while (table_iterator != std::cbegin(dp_table)){
212  min_ic = std::min_element(std::cbegin(*table_iterator), std::cend(*table_iterator));
213  std::size_t position = std::distance(std::cbegin(*table_iterator), min_ic);
