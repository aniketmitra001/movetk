79  TrajectoryInputIterator _traj_it;
80  TrajectoryInputIterator _traj_it_end;
81  double _time_diff_threshold_s;
82  double _distance_threshold_m;
83  std::size_t _min_points; // min number of points required for a split to qualify as a trajectory
84 
85  // iteration state
86  std::vector<trajectory_type> _splits;
87  typename std::vector<trajectory_type>::iterator _split_it;
88 
89  inline std::optional<trajectory_type> read_trajectory()
90  {
91  if (_traj_it != _traj_it_end && _split_it == std::end(_splits)) {
92  _splits.clear(); // reset for next use
93 
94  while(_splits.size() == 0 && _traj_it != _traj_it_end) {
95  // std::cout << "Splitting... " << std::get<0>(_traj_it->data()[0]) << '\n';
96 
97  // move to the next trajectory, split it and return the first split segment
98  using SplitByTimeDiff = SplitByDifferenceThreshold<DateIdx, ProbePoint>;
99  SplitByTimeDiff split_by_time_diff(_time_diff_threshold_s);
100  using SplitByDistance = SplitByDistanceThreshold<LatIdx, LonIdx, ProbePoint>;
101  std::function<double(float, float, float, float)> distancefn = distance_exact;
102  SplitByDistance split_by_dist(_distance_threshold_m, distancefn);
103  auto split_by_time_diff_or_distance = [&](const ProbePoint& p) {
104  bool t = split_by_time_diff(p);
105  bool d = split_by_dist(p);
106  return (t || d);
107  };
108 
109  using ProbeInputIterator = decltype(_traj_it->begin());
110  Splitter<decltype(split_by_time_diff_or_distance), ProbeInputIterator> hifreq_splitter(_traj_it->begin(),
111  _traj_it->end(), split_by_time_diff_or_distance);
112 
113 
114  for (auto& segment: hifreq_splitter) {
115  if (segment.size() >= _min_points)
116  _splits.emplace_back(trajectory_type { segment });
117  }
118  // std::cout << "Got " << _splits.size() << " splits\n";
119  _split_it = std::begin(_splits);
120  ++_traj_it;
121  }
122  }
123 
124  if (_split_it != std::end(_splits)) {
125  trajectory_type t = *_split_it;
126  ++_split_it;
127  return t;
128  }
129  else
130  return std::nullopt;
131  }
