90 double euclidean_distance(double x0, double y0, double x1, double y1)
91 {
92  auto dx = x1 - x0;
93  auto dy = y1 - y0;
94  return sqrt(dx * dx + dy * dy);
95 }

97 double euclidean_distance_3d(double x0, double y0, double z0, double x1, double y1, double z1)
98 {
99  auto dx = x1 - x0;
100  auto dy = y1 - y0;
101  auto dz = z1 - z0;
