44 public:
45  coup_mat(size_type1 w, size_type2 h)
46  : m_data(w * h,-1), m_width(w), m_height(h)
47  {}
48 
49  result_type & operator()(size_type1 i, size_type2 j)

61 struct linestring_linestring
62 {
63  template <typename Linestring1, typename Linestring2, typename Strategy>
64  static inline typename distance_result
65  <
66  typename point_type<Linestring1>::type,
67  typename point_type<Linestring2>::type,
68  Strategy
69  >::type apply(Linestring1 const& ls1, Linestring2 const& ls2, Strategy const& strategy)
70  {
71  typedef typename distance_result
72  <
73  typename point_type<Linestring1>::type,
74  typename point_type<Linestring2>::type,
75  Strategy
76  >::type result_type;
77  typedef typename boost::range_size<Linestring1>::type size_type1;
78  typedef typename boost::range_size<Linestring2>::type size_type2;
79 
80 
81  boost::geometry::detail::throw_on_empty_input(ls1);
82  boost::geometry::detail::throw_on_empty_input(ls2);
83 
84  size_type1 const a = boost::size(ls1);
85  size_type2 const b = boost::size(ls2);
86 
87 
88  //Coupling Matrix CoupMat(a,b,-1);
89  coup_mat<size_type1,size_type2,result_type> coup_matrix(a,b);
90 
91  result_type const not_feasible = -100;
92  //findin the Coupling Measure
93  for(size_type1 i=0;i<a;i++)
94  {
95  for(size_type2 j=0;j<b;j++)
96  {
97  result_type dis = geometry::distance(range::at(ls1,i),range::at(ls2,j),strategy);
98  if(i==0 && j==0)
99  coup_matrix(i,j)= dis;
100  else if(i==0 && j>0)
101  coup_matrix(i,j)=
102  (std::max)(coup_matrix(i,j-1),dis);
103  else if(i>0 && j==0)
104  coup_matrix(i,j)=
105  (std::max)(coup_matrix(i-1,j),dis);
106  else if(i>0 && j>0)
107  coup_matrix(i,j)=
108  (std::max)((std::min)(coup_matrix(i,j-1),(std::min)(coup_matrix(i-1,j),coup_matrix(i-1,j-1))),dis);
109  else
110  coup_matrix(i,j)=not_feasible;
111  }
112  }
113 
114  #ifdef BOOST_GEOMETRY_DEBUG_FRECHET_DISTANCE
115  //Print CoupLing Matrix
116  for(size_type i = 0; i <a; i++)
117  {
118  for(size_type j = 0; j <b; j++)
119  std::cout << coup_matrix(i,j) << " ";
120  std::cout << std::endl;
121  }
122  #endif
123 
124  return coup_matrix(a-1,b-1);
125  }
126 };
