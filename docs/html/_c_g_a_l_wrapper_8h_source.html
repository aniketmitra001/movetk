853  template<typename PointIterator,
854  typename = movetk_core::requires_random_access_iterator<PointIterator>,
855  typename = movetk_core::requires_wrapper_point<Kernel,
856  typename PointIterator::value_type >>
857  void to_CGAL_points(PointIterator first, PointIterator beyond) {
858  //ASSERT_RANDOM_ACCESS_ITERATOR(PointIterator);
859  // ASSERT_WRAPPER_POINT_TYPE(Kernel, first);
860  PointIterator it = first;
861  while (it != beyond) {
862  points.insert((*it).get());
863  it++;
864  }

867  public:
881  template<class PointIterator, class CenterIterator,
882  typename = movetk_core::requires_random_access_iterator<PointIterator>,
883  typename = movetk_core::requires_wrapper_point<Kernel,
884  typename PointIterator::value_type>,
885  typename = movetk_core::requires_output_iterator<CenterIterator>,
886  typename = movetk_core::requires_NT<Kernel,
887  typename CenterIterator::value_type> >
888  NT operator()(PointIterator first, PointIterator beyond, CenterIterator iter) {
889  //ASSERT_RANDOM_ACCESS_ITERATOR(PointIterator);
890  //ASSERT_WRAPPER_POINT_TYPE(Kernel, first);
891  //ASSERT_OUTPUT_ITERATOR(CenterIterator);
892  this->to_CGAL_points(first, beyond);
893  CGAL_MinSphere ms(begin(points), end(points));
894  radius = ms.radius().first +
895  ms.radius().second * sqrt(ms.discriminant());
896  for (typename CGAL_MinSphere::Cartesian_const_iterator cit =
897  ms.center_cartesian_begin(); cit != ms.center_cartesian_end();
898  cit++) {
899  *iter = cit->first + cit->second;
900  }
901  points.clear();
902  return radius;
903  }

932  template<class Kernel>
933  class Wrapper_Curve_Intersection {
934  private:
935  typedef typename Kernel::Wrapper_CGAL_Kernel_2::Point_2 Point_2;
936  typedef typename Kernel::Wrapper_CGAL_Kernel_2::Curve_2 Curve;
937  typedef Wrapper_CGAL_Point<Kernel> Wrapper_Point;
938  typedef Wrapper_CGAL_Vector<Kernel> Wrapper_Vector;
939  std::vector<Curve> Curves;
940  std::list<Point_2> Points;
941  public:
942  template <class InputIterator>
943  std::size_t operator()(InputIterator first, InputIterator beyond){
944  InputIterator it = first + 1;
945  Curves.clear();
946  Points.clear();
947  while (it != beyond){
948  Wrapper_Point p1 = *(it - 1);
949  Wrapper_Point p2 = *it;
950  Wrapper_Vector v = p2 - p1;
951  if ((v * v) < MOVETK_EPS) {
952  it++;
953  continue;
954  }
955  auto CoordinateIterP1 = p1.begin();
956  auto CoordinateIterP2 = p2.begin();
957  Curves.push_back(Curve(Point_2(*CoordinateIterP1, *(CoordinateIterP1 + 1)),
958  Point_2(*CoordinateIterP2, *(CoordinateIterP2 + 1))
959  )
960  );
961  it++;
962  }
963  CGAL::compute_intersection_points(std::begin(Curves),
964  std::end(Curves),std::back_inserter(Points));
965  return Points.size();
966  }
