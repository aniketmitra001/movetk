126  template<class InputIterator, class Container,
127  typename = movetk_core::requires_random_access_iterator<InputIterator>,
128  typename = movetk_core::requires_equality<
129  InputIterator,
130  typename Container::value_type::value_type> >
131  typename Container::const_iterator operator()(InputIterator first, InputIterator beyond, Container& sequences){
132  auto it = first;
133  movetk_core::movetk_back_insert_iterator bit(sequences);
134  for (; it != beyond; it++)
135  {
136  bool extend_subsequence = false;
137  for (auto sit = std::begin(sequences); sit != std::end(sequences); sit++)
138  {
139  InputIterator prev = *(sit->cend() - 1);
140  if (test(*prev, *it))
141  {
142  movetk_core::movetk_back_insert_iterator node_bit(*sit);
143  node_bit = it;
144  extend_subsequence = true;
145  }
146  }
147  if (!extend_subsequence)
148  {
149  typename Container::value_type node;
150  movetk_core::movetk_back_insert_iterator node_bit(node);
151  node_bit = it;
152  bit = node;
153  }
154  }
155 
156  std::sort(std::begin(sequences), std::end(sequences), [](auto a, auto b){
157  std::size_t size_a = std::distance(std::begin(a), std::end(a));
158  std::size_t size_b = std::distance(std::begin(b), std::end(b));
159  return size_a > size_b;
160  });
161 
162  auto sit = std::cbegin(sequences);
163  std::size_t max_size = sit->size();
164 
165  auto eit = std::remove_if(std::begin(sequences), std::end(sequences), [max_size](auto a){
166  std::size_t size_a = std::distance(std::begin(a), std::end(a));
167  return size_a != max_size;
168  });
169 
170  return eit;
171 
172  } //operator()

201  template <class InputIterator, class OutputIterator,
202  typename = movetk_core::requires_random_access_iterator<InputIterator>,
203  typename = movetk_core::requires_output_iterator<OutputIterator>,
204  typename = movetk_core::requires_equality<typename InputIterator::value_type,
205  typename OutputIterator::value_type::value_type >>
206  void operator()(InputIterator first, InputIterator beyond, OutputIterator result)
207  {
208  vector<size_t> segments;
209  size_t segment_size = 1;
210  auto it = first;
211  auto prev = *it;
212  it++;
213  for (; it != beyond; it++)
214  {
215  if (test(prev, *it))
216  segment_size++;
217  else
218  {
219  segments.push_back(segment_size);
220  segment_size = 1;
221  }
222  prev = *it;
223  }
224  segments.push_back(segment_size);
225  it = first;
226  for (auto segment : segments)
227  {
228  if (segment >= min_seg_size)
229  for (size_t i = 0; i < segment; i++)
230  {
231  *result = it;
232  it++;
233  }
234  else it = next(it, segment);
235  }
236  } //operator()
237  };

431  template <class InputIterator, class OutputIterator,
432  typename = movetk_core::requires_random_access_iterator<InputIterator>,
433  typename = movetk_core::requires_output_iterator<OutputIterator>,
434  typename = movetk_core::requires_equality<typename InputIterator::value_type,
435  typename OutputIterator::value_type::value_type >>
436  void operator()(InputIterator first, InputIterator beyond, OutputIterator result)
437  {
438  InputIterator it = first;
439  Container<InputIterator> list(it);
440  while(++it != beyond){
441  bool consistent = false;
442  auto iter = list.end();
443  auto longest_sequence = list.begin();
444  std::size_t longest_sequence_length = longest_sequence->sequence_length;
445  while (iter != list.beyond()){
446  if (test(*it, *iter->value)){
447  std::size_t sequence_length = iter->sequence_length;
448  if ( longest_sequence_length < sequence_length ){
449  longest_sequence = iter;
450  longest_sequence_length = iter->sequence_length + 1;
451  }
452  consistent = true;
453  }
454  iter--;
455  }
456  if (consistent){
457  list.insert(longest_sequence, it);
458  }
459  if (!consistent){
460  list.insert(it);
461  }
462  }
463  auto iter = list.last();
464  while (iter != list.beyond()){
465  result = iter->value;
466  iter--;
467  }
468  }
469  };
