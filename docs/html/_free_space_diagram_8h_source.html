86  template<class InputIterator>
87  void identify_free_vertices(InputIterator first, InputIterator beyond,
88  typename CellTraits::GeometryTraits::MovetkSegment S,
89  std::size_t edge_id) {
90 
91  if ((edge_id != CellTraits::edge_orientation::Left) &&
92  (edge_id != CellTraits::edge_orientation::Right))
93  return;
94 
95  std::size_t num_intersections = std::distance(first, beyond);
96 
97  if (num_intersections == 1) {
98  auto value = std::get<Attributes::SQUARED_RATIO>(*first);
99  if ((value > 0) && (value < 1))
100  return;
101  if (value == 1) {
102  if (edge_id == CellTraits::edge_orientation::Left)
103  vertices.insert(CellTraits::vertex_orientation::TopLeft);
104  if (edge_id == CellTraits::edge_orientation::Right)
105  vertices.insert(CellTraits::vertex_orientation::TopRight);
106  }
107  if (value == 0) {
108  if (edge_id == CellTraits::edge_orientation::Left)
109  vertices.insert(CellTraits::vertex_orientation::BottomLeft);
110  if (edge_id == CellTraits::edge_orientation::Right)
111  vertices.insert(CellTraits::vertex_orientation::BottomRight);
112  }
113  }
114 
115  if (num_intersections == 2) {
116  auto sign_ip_1 = std::get<Attributes::SIGN_DISCRIMINANT>(*first);
117  auto sign_ip_2 = std::get<Attributes::SIGN_DISCRIMINANT>(*(first + 1));
118 
119  if ((sign_ip_1 == 1) && (sign_ip_2 == 1))
120  return;
121 
122  typename CellTraits::GeometryTraits::MovetkPoint ip_1 = std::get<Attributes::POINT>(*first);
123  typename CellTraits::GeometryTraits::MovetkPoint ip_2 = std::get<Attributes::POINT>(*(first + 1));
124  std::size_t orientation_ip_1 = CellTraits::vertex_orientation::BottomLeft;
125  std::size_t orientation_ip_2 = CellTraits::vertex_orientation::BottomRight;
126  if (sign_ip_1 == -1) {
127  typename CellTraits::GeometryTraits::MovetkVector v1 = S[0] - ip_1;
128  typename CellTraits::GeometryTraits::MovetkVector v2 = S[1] - ip_1;
129  if ((v1 * v1) < MOVETK_EPS) {
130  if (edge_id == CellTraits::edge_orientation::Left)
131  orientation_ip_1 = CellTraits::vertex_orientation::BottomLeft;
132  if (edge_id == CellTraits::edge_orientation::Right)
133  orientation_ip_1 = CellTraits::vertex_orientation::BottomRight;
134  } else if ((v2 * v2) < MOVETK_EPS) {
135  if (edge_id == CellTraits::edge_orientation::Left)
136  orientation_ip_1 = CellTraits::vertex_orientation::TopLeft;
137  if (edge_id == CellTraits::edge_orientation::Right)
138  orientation_ip_1 = CellTraits::vertex_orientation::TopRight;
139  }
140  }
141  if (sign_ip_2 == -1) {
142  typename CellTraits::GeometryTraits::MovetkVector v1 = S[0] - ip_2;
143  typename CellTraits::GeometryTraits::MovetkVector v2 = S[1] - ip_2;
144  if ((v1 * v1) < MOVETK_EPS) {
145  if (edge_id == CellTraits::edge_orientation::Left)
146  orientation_ip_2 = CellTraits::vertex_orientation::BottomLeft;
147  if (edge_id == CellTraits::edge_orientation::Right)
148  orientation_ip_2 = CellTraits::vertex_orientation::BottomRight;
149  } else if ((v2 * v2) < MOVETK_EPS) {
150  if (edge_id == CellTraits::edge_orientation::Left)
151  orientation_ip_2 = CellTraits::vertex_orientation::TopLeft;
152  if (edge_id == CellTraits::edge_orientation::Right)
153  orientation_ip_2 = CellTraits::vertex_orientation::TopRight;
154  }
155  }
156 
157  if ((sign_ip_1 == -1) && (sign_ip_2 == -1)) {
158  if (orientation_ip_1 == orientation_ip_2) {
159  return;
160  } else {
161  vertices.insert(orientation_ip_1);
162  vertices.insert(orientation_ip_2);
163  }
164  } else if ((sign_ip_1 == -1) && (sign_ip_2 != -1)) {
165  vertices.insert(orientation_ip_1);
166  } else if ((sign_ip_1 != -1) && (sign_ip_2 == -1)) {
167  vertices.insert(orientation_ip_2);
168  }
169 
170  }

183  FreeSpaceCell(typename CellTraits::GeometryTraits::MovetkSegment P,
184  typename CellTraits::GeometryTraits::MovetkSegment Q,
185  typename CellTraits::GeometryTraits::NT radius) {
186 
187  typename CellTraits::GeometryTraits::MovetkSphere sphere = make_sphere(Q[0], radius);
188  std::size_t init_length = std::distance(std::begin(intersections), std::end(intersections));
189  compute_intersections(sphere, P, typename CellTraits::OutputIterator(intersections));
190  std::size_t new_length = std::distance(std::begin(intersections), std::end(intersections));
191  update_edge_id(std::begin(intersections) + init_length, std::end(intersections),
192  CellTraits::edge_orientation::Left);
193  identify_free_vertices(std::begin(intersections) + init_length, std::end(intersections),
194  P, CellTraits::edge_orientation::Left);
195  init_length = new_length;
196 
197  sphere = make_sphere(P[1], radius);
198  compute_intersections(sphere, Q, typename CellTraits::OutputIterator(intersections));
199  new_length = std::distance(std::begin(intersections), std::end(intersections));
200  update_edge_id(std::begin(intersections) + init_length, std::end(intersections),
201  CellTraits::edge_orientation::Top);
202  init_length = new_length;
203 
204  sphere = make_sphere(Q[1], radius);
205  compute_intersections(sphere, P, typename CellTraits::OutputIterator(intersections));
206  new_length = std::distance(std::begin(intersections), std::end(intersections));
207  update_edge_id(std::begin(intersections) + init_length, std::end(intersections),
208  CellTraits::edge_orientation::Right);
209  identify_free_vertices(std::begin(intersections) + init_length, std::end(intersections),
210  P, CellTraits::edge_orientation::Right);;
211  init_length = new_length;
212 
213  sphere = make_sphere(P[0], radius);
214  compute_intersections(sphere, Q, typename CellTraits::OutputIterator(intersections));
215  new_length = std::distance(std::begin(intersections), std::end(intersections));
216  update_edge_id(std::begin(intersections) + init_length,
217  std::end(intersections), CellTraits::edge_orientation::Bottom);
218  init_length = new_length;
219 
220  }

276  template<class InputIterator>
277  FreeSpaceDiagram(InputIterator polyline_p_first, InputIterator polyline_p_beyond,
278  InputIterator polyline_q_first, InputIterator polyline_q_beyond,
279  typename FreeSpaceDiagramTraits::GeometryTraits::NT radius) {
280  auto rit = typename FreeSpaceDiagramTraits::RowsOutputIterator(rows);
281  auto pit = polyline_p_first;
282  std::size_t num_cols = std::distance(polyline_q_first, polyline_q_beyond);
283  std::size_t num_rows = std::distance(polyline_p_first, polyline_p_beyond);
284  std::size_t row_idx = 0;
285  while (pit != (polyline_p_beyond - 1)) {
286  auto qit = polyline_q_first;
287  typename FreeSpaceDiagramTraits::Rows::value_type cells;
288  auto cit = typename FreeSpaceDiagramTraits::CellsOutputIterator(cells);
289  std::size_t col_idx = 0;
290  while (qit != (polyline_q_beyond - 1)) {
291  typename FreeSpaceDiagramTraits::GeometryTraits::MovetkSegment P = make_segment(*pit, *(pit + 1));
292  typename FreeSpaceDiagramTraits::GeometryTraits::MovetkSegment Q = make_segment(*qit, *(qit + 1));
293  typename FreeSpaceDiagramTraits::Cell fsc(P, Q, radius);
294  cit = fsc;
295  qit++;
296  col_idx++;
297  }
298  rit = cells;
299  row_idx++;
300  pit++;
301  }
302 
303  }
304  typename FreeSpaceDiagramTraits::iterator begin() {
305  return typename FreeSpaceDiagramTraits::iterator(std::begin(rows));
306  }
